# Handle subtype dropdown selection
DI_artifact_subtype_select = {
    scope = character # Player Character

    saved_scopes = {
        subtype # Selected subtype flag
    }

    effect = {
        save_scope_as = player

        # Update display variable to match the selected subtype flag
        set_variable = {
            name = DI_artifact_subtype_display
            value = scope:subtype
        }

        # Set the actual artifact type variable (map display flag to internal type flag)
        if = {
            limit = { scope:subtype = flag:DI_subtype_axe }
            set_variable = { name = DI_artifact_type value = flag:axe }

# Handle quality dropdown selection
DI_artifact_quality_select = {
    scope = character # Player Character

    saved_scopes = {
        quality # Selected quality flag
    }

    effect = {
        save_scope_as = player

        # Update display variable to match the selected quality flag
        set_variable = {
            name = DI_artifact_quality_display
            value = scope:quality
        }

        # Set the actual quality variable (map display flag to internal quality flag)
        if = {
            limit = { scope:quality = flag:DI_quality_common }
            set_variable = { name = DI_artifact_quality value = flag:common }
        }
        else_if = {
            limit = { scope:quality = flag:DI_quality_masterwork }
            set_variable = { name = DI_artifact_quality value = flag:masterwork }
        }
        else_if = {
            limit = { scope:quality = flag:DI_quality_famed }
            set_variable = { name = DI_artifact_quality value = flag:famed }
        }
        else_if = {
            limit = { scope:quality = flag:DI_quality_illustrious }
            set_variable = { name = DI_artifact_quality value = flag:illustrious }
        }
    }
}
