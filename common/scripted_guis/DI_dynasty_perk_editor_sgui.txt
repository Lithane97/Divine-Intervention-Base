### Select Character ###
DI_dynasty_selected_char = {
    scope = character

    saved_scopes = {
        target # player character
    }

    effect = {
        save_scope_as = select_char

        scope:target = {
            set_variable = {
                name = DI_dynasty_selected_char
                value = scope:select_char

            }
            set_variable = {
                name = DI_dynasty_selected_dynasty
                value = scope:select_char.dynasty
            }
        }
        debug_log = "Selected dynasty edit character set"
        debug_log = "[scope:select_char.GetFullName]"
    }
}

                }
            }
        }

}

### Select Character Init ###
DI_dynasty_perk_select_char_init = {
    scope = character

    effect = {
        if = {
            limit = {
                NOT = { has_variable = DI_dynasty_selected_char }
            }
            set_variable = {
                name = DI_dynasty_selected_char
                value = this
            }
            set_variable = {
                name = DI_dynasty_selected_dynasty
                value = this.dynasty
            }
        }
        debug_log = "Initialized dynasty edit character set"
        debug_log = "[scope:this.GetFullName]"
    }
}
