### Select Character Init ###
DI_artifact_select_char_init = {
    scope = character

    effect = {
        if = {
            limit = {
                NOT = { has_variable = DI_artifact_selected_char }
            }
            set_variable = {
                name = DI_artifact_selected_char
                value = this
            }
        }
    }
}

DI_take_artifact = {
    scope = character

    saved_scopes = {
        selected_artifact # artifact
    }

    effect = {
        save_scope_as = new_owner

        scope:selected_artifact = {
            set_owner = scope:new_owner
        }
    }
}

DI_destroy_artifact = {
    scope = character

    saved_scopes = {
        selected_artifact # artifact
    }

    effect = {
        scope:selected_artifact = { save_scope_as = selected_artifact }
        destroy_artifact = scope:selected_artifact
    }
}

DI_get_artifact_list = {
    scope = character

    effect = {
        save_scope_as = target

        every_in_list = {
            variable = pinned_characters

            every_character_artifact  = {
                scope:target = {
                    add_to_variable_list = { name = DI_artifact_list target = prev }
                }
            }
        }
    }
}

DI_clear_artifact_list = {
    scope = character

    effect = {
        clear_variable_list = DI_artifact_list
    }
}

### Artifact Creation Scripts ###
DI_create_artifact = {
    scope = character

    effect = {
        save_scope_as = player

        var:DI_artifact_selected_char = {
            # Set rarity bonus
            DI_set_rarity_bonus = yes

            # Get artifact parameters based on type
            DI_get_artifact_params = yes

            # Route to appropriate creation effect based on category
            if = {
                limit = { var:artifact_category = flag:weapon }
                DI_create_weapon_internal = yes
            }
            else_if = {
                limit = { var:artifact_category = flag:armor }
                DI_create_armor_internal = yes
            }
            else_if = {
                limit = { var:artifact_category = flag:regalia }
                DI_create_regalia_internal = yes
            }
            else_if = {
                limit = { var:artifact_category = flag:trinket }
                DI_create_trinket_internal = yes
            }
            else_if = {
                limit = { var:artifact_category = flag:throne }
                DI_create_throne_internal = yes
            }
            else_if = {
                limit = {
                    OR = {
                        var:artifact_category = flag:pedestal
                        var:artifact_category = flag:furniture
                        var:artifact_category = flag:ornament
                    }
                }
                DI_create_court_artifact_internal = yes
            }
        }
    }
}
