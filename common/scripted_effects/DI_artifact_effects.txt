# Convert quality flag to rarity string for create_artifact
DI_get_rarity_string = {
    # Returns the rarity as a string that can be used in create_artifact
    # Must be called in player scope
    if = { limit = { var:DI_artifact_quality = flag:common } save_scope_value_as = { name = rarity_string value = flag:common } }
    else_if = { limit = { var:DI_artifact_quality = flag:masterwork } save_scope_value_as = { name = rarity_string value = flag:masterwork } }
    else_if = { limit = { var:DI_artifact_quality = flag:famed } save_scope_value_as = { name = rarity_string value = flag:famed } }
    else_if = { limit = { var:DI_artifact_quality = flag:illustrious } save_scope_value_as = { name = rarity_string value = flag:illustrious } }
}

# Set quality and wealth scopes based on selected rarity
DI_set_rarity_bonus = {
    # Quality ranges: 100-76 (illustrious), 75-51 (famed), 50-26 (masterwork), 25-0 (common)
    # Wealth ranges: 85+ (illustrious), 55-84 (famed), 25-54 (masterwork), 0-24 (common)
    # Random values are weighted toward higher end of range

    if = {
        limit = { scope:player.var:DI_artifact_quality = flag:common }
        # Quality: 0-25 range
        random_list = {
            50 = { save_scope_value_as = { name = quality value = 20 } }  # Higher end
            30 = { save_scope_value_as = { name = quality value = 15 } }
            15 = { save_scope_value_as = { name = quality value = 10 } }
            5 = { save_scope_value_as = { name = quality value = 5 } }    # Lower end
        }
        # Wealth: 0-24 range
        random_list = {
            50 = { save_scope_value_as = { name = wealth value = 20 } }   # Higher end
            30 = { save_scope_value_as = { name = wealth value = 15 } }
            15 = { save_scope_value_as = { name = wealth value = 10 } }
            5 = { save_scope_value_as = { name = wealth value = 5 } }     # Lower end
        }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:masterwork }
        # Quality: 26-50 range
        random_list = {
            50 = { save_scope_value_as = { name = quality value = 45 } }  # Higher end
            30 = { save_scope_value_as = { name = quality value = 38 } }
            15 = { save_scope_value_as = { name = quality value = 32 } }
            5 = { save_scope_value_as = { name = quality value = 26 } }   # Lower end
        }
        # Wealth: 25-54 range
        random_list = {
            50 = { save_scope_value_as = { name = wealth value = 50 } }   # Higher end
            30 = { save_scope_value_as = { name = wealth value = 42 } }
            15 = { save_scope_value_as = { name = wealth value = 34 } }
            5 = { save_scope_value_as = { name = wealth value = 26 } }    # Lower end
        }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:famed }
        # Quality: 51-75 range
        random_list = {
            50 = { save_scope_value_as = { name = quality value = 70 } }  # Higher end
            30 = { save_scope_value_as = { name = quality value = 64 } }
            15 = { save_scope_value_as = { name = quality value = 58 } }
            5 = { save_scope_value_as = { name = quality value = 52 } }   # Lower end
        }
        # Wealth: 55-84 range
        random_list = {
            50 = { save_scope_value_as = { name = wealth value = 78 } }   # Higher end
            30 = { save_scope_value_as = { name = wealth value = 70 } }
            15 = { save_scope_value_as = { name = wealth value = 62 } }
            5 = { save_scope_value_as = { name = wealth value = 56 } }    # Lower end
        }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:illustrious }
        # Quality: 76-100 range
        random_list = {
            50 = { save_scope_value_as = { name = quality value = 95 } }  # Higher end
            30 = { save_scope_value_as = { name = quality value = 88 } }
            15 = { save_scope_value_as = { name = quality value = 82 } }
            5 = { save_scope_value_as = { name = quality value = 76 } }   # Lower end
        }
        # Wealth: 85+ range
        random_list = {
            50 = { save_scope_value_as = { name = wealth value = 98 } }   # Higher end
            30 = { save_scope_value_as = { name = wealth value = 92 } }
            15 = { save_scope_value_as = { name = wealth value = 88 } }
            5 = { save_scope_value_as = { name = wealth value = 85 } }    # Lower end
        }
    }
}

DI_add_new_modifiers_effect = {
    # Prowess from weapon rarity
    add_scaled_artifact_modifier_prowess_effect = yes

    if = {
        limit = {
            OR = {
                rarity = famed
                rarity = illustrious
            }
            OR = {
                has_artifact_feature_group = blade_decoration
                has_artifact_feature_group = hilt_decoration
                has_artifact_feature_group = shaft_decoration
                has_artifact_feature_group = head_decoration
                scope:weapon_type = flag:artifact_weapon_type_dagger
            }
        }
        if = {
            limit = { scope:weapon_type = flag:artifact_weapon_type_dagger }
            add_2_scaled_artifact_modifier_intrigue_effect = yes
        }
        else = {
            add_2_scaled_artifact_modifier_combat_effect = yes
        }
    }
    else_if = {
        limit = {
            OR = {
                rarity = common
                rarity = masterwork
            }
            OR = {
                has_artifact_feature_group = blade_decoration
                has_artifact_feature_group = hilt_decoration
                has_artifact_feature_group = shaft_decoration
                has_artifact_feature_group = head_decoration
                scope:weapon_type = flag:artifact_weapon_type_dagger
            }
        }
        if = {
            limit = { scope:weapon_type = flag:artifact_weapon_type_dagger }
            add_scaled_artifact_modifier_intrigue_effect = yes
        }
        else = {
            add_scaled_artifact_modifier_combat_effect = yes
        }
    }
    else_if = {
        limit = {
            OR = {
                rarity = famed
                rarity = illustrious
            }
            # No decorations
        }
        if = {
            limit = { scope:weapon_type = flag:artifact_weapon_type_dagger }
            add_scaled_artifact_modifier_intrigue_effect = yes
            add_artifact_modifier = artifact_prowess_no_decorations_high_tier_modifier
        }
        else = {
            add_scaled_artifact_modifier_combat_effect = yes
            add_artifact_modifier = artifact_prowess_no_decorations_high_tier_modifier
        }
    }
    else_if = {
        limit = {
            rarity = masterwork
            # No decorations
        }
        if = {
            limit = { scope:weapon_type = flag:artifact_weapon_type_dagger }
            add_scaled_artifact_modifier_intrigue_effect = yes
            add_artifact_modifier = artifact_prowess_no_decorations_modifier
        }
        else = {
            add_artifact_modifier = artifact_prowess_no_decorations_medium_tier_modifier
        }
    }
    else = {
        add_artifact_modifier = artifact_prowess_no_decorations_modifier
    }

}

DI_add_new_armor_modifiers_effect = {
    # Rarity determines prowess bonus
    add_scaled_artifact_modifier_prowess_effect = yes
    # Rarity determines combat bonus
    add_scaled_artifact_modifier_combat_effect = yes

    if = { # Chain mail gets a higher prowess due to the lack of decoration
        limit = { artifact_type = armor_mail }
        add_artifact_modifier = artifact_prowess_no_decorations_modifier
    }
    else_if = { # Fashion gets an attraction boost
        limit = {
            OR = {
                artifact_type = armor_brigandine
                artifact_type = armor_plate
            }
        }
        add_artifact_modifier = artifact_attraction_opinion_1_modifier
    }
    else = { # For all other armor-types you get some prestige
        add_artifact_modifier = artifact_placeholder_modifier
    }

    if = { # Gold gives more prestige
        limit = {
            OR = {
                has_artifact_feature = armor_decoration_gold_plating
                has_artifact_feature = armor_material_gold
            }
        }
        add_artifact_modifier = artifact_monthly_prestige_2_modifier
    }

}

# Internal weapon creation (called from DI_create_artifact_unified)
DI_create_weapon_internal = {
    save_scope_as = owner

    # Set weapon type for modifier application
    set_variable = { name = artifact_weapon_type value = var:artifact_subtype }
    scope:owner = {
        var:artifact_weapon_type = { save_scope_as = weapon_type }
        remove_variable = artifact_weapon_type
    }

    # Create with appropriate rarity
    if = {
        limit = { scope:player.var:DI_artifact_quality = flag:common }
        DI_create_weapon_rarity = { RARITY = common }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:masterwork }
        DI_create_weapon_rarity = { RARITY = masterwork }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:famed }
        DI_create_weapon_rarity = { RARITY = famed }
    }
    else_if = {
        limit = { scope:player.var:DI_artifact_quality = flag:illustrious }
        DI_create_weapon_rarity = { RARITY = illustrious }
    }
}

DI_create_weapon_rarity = {
    # Helper to route to final weapon creation with double parameter injection
    if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_axe }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = axe WEAPON_NAME = artifact_axe_name WEAPON_DESC = artifact_axe_description DECORATION_EFFECT = get_axe_decoration_effect }
    }
    else_if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_dagger }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = dagger WEAPON_NAME = artifact_dagger_name WEAPON_DESC = artifact_dagger_description DECORATION_EFFECT = get_dagger_decoration_effect }
    }
    else_if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_hammer }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = hammer WEAPON_NAME = artifact_hammer_name WEAPON_DESC = artifact_hammer_description DECORATION_EFFECT = get_hammer_decoration_effect }
    }
    else_if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_mace }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = mace WEAPON_NAME = artifact_mace_name WEAPON_DESC = artifact_mace_description DECORATION_EFFECT = get_mace_decoration_effect }
    }
    else_if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_spear }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = spear WEAPON_NAME = artifact_spear_name WEAPON_DESC = artifact_spear_description DECORATION_EFFECT = get_spear_decoration_effect }
    }
    else_if = {
        limit = { var:artifact_subtype = flag:artifact_weapon_type_sword }
        DI_create_weapon_final = { RARITY = $RARITY$ WEAPON_TYPE = sword WEAPON_NAME = artifact_sword_name WEAPON_DESC = artifact_sword_description DECORATION_EFFECT = get_sword_decoration_effect }
    }
}

DI_create_weapon_final = {
    # Final weapon creation with multiple parameter injection
    create_artifact = {
        name = $WEAPON_NAME$
        description = placeholder
        rarity = $RARITY$
        type = $WEAPON_TYPE$
        visuals = $WEAPON_TYPE$
        modifier = artifact_prowess_1_modifier
        generate_history = yes
        creator = root
        wealth = scope:wealth
        quality = scope:quality
        save_scope_as = newly_created_artifact
    }

    scope:newly_created_artifact = {
        $DECORATION_EFFECT$ = yes
        get_artifact_feature_references_effect = yes
        set_artifact_description = $WEAPON_DESC$
        DI_add_new_modifiers_effect = yes
    }
}

# Minimal deterministic animal hide creation for testing.
# - No hunt randomization / location logic.
# - You control animal + size.
#
# Required injections you can hardcode for a test button:
#   scope:owner        -> scope:some_character (or root)
#   scope:creator      -> root (or whoever)
#   $RARITY$       -> common / masterwork / famed / illustrious ...
#   scope:animal      -> flag:stag / flag:boar / flag:wolf / flag:roe / flag:bear ...

DI_create_animal_hide_test = {
    # Choose a location scope for the description to reference.
    # If your injected hunter has a .location, prefer that; otherwise fall back to owner.location.
    if = {
        limit = { exists = scope:hunter }
        scope:hunter.location = { save_scope_as = location }
    }
    else = {
        scope:owner.location = { save_scope_as = location }
    }

    # Seed the variable that animal_hide_type_* triggers read.
    scope:owner = {
        set_variable = { name = animal_type value = scope:animal days = 5 }
    }

    create_artifact = {
        name = artifact_animal_hide_name_base
        creator = scope:creator
        rarity = $RARITY$
        description = placeholder
        visuals = large_animal_hide
        type = animal_hide_big
        modifier = artifact_placeholder_modifier
        wealth = scope:wealth
        quality = scope:quality
        save_scope_as = newly_created_artifact
    }

    scope:newly_created_artifact = {
        # Debug / introspection
        set_variable = { name = animal_type value = scope:owner.var:animal_type }
        set_artifact_feature = animal_hide_type_bear
        # Force-pick the required hide feature group.
        # set_artifact_feature_group = animal_hide_type


        # These two are what your description string is almost certainly missing:
        # (Vanilla uses "adventure_artifact_hunter" and "adventure_artifact_location")
        set_variable = { name = adventure_artifact_hunter value = scope:hunter }
        set_variable = { name = adventure_artifact_location value = scope:location }

        # Fill common reference groups (patterns/material refs/etc.) if needed
        get_artifact_feature_references_effect = yes

        remove_artifact_modifier = artifact_placeholder_modifier

        # If you want to keep vanilla-ish hide description:
        set_artifact_description = artifact_animal_hide_description
        set_artifact_name = artifact_animal_hide_name_base
    }

    # Cleanup
    scope:owner = {
        remove_variable = animal_type
    }
}
